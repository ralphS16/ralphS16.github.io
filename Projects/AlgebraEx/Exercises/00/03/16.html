My Python code (you need to define the $\gcd$ function as in 0.2.9): 

def mod(a,n):
    #returns the integer 0<= c < n such that a = c (mod n)
    return a - (a/n)*n

def add(a,b,n):
    #adds a and b in the ring Z/nZ (returns a+b (mod n))
    return mod(a+b,n)

def mult(a,b,n):
    #returns a*b (mod n)
    return mod(a*b, n)

def inverse(a,n):
    return mod(int(gcd(n,a)[2]),n)
